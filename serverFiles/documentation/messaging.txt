--message api documentation-- (by Gloshayden)

terms:
    id: random numbers, the id of the chat
    owner username: can be the username of an account or "" for all users in a chat
    chats dictionary:
        {
            "name":name,
            "id":id,
            "owner":owner,
            "users":[user],
            "messages":[]
        }

api uses:
    "message-addUser": Adds a user to the chat
    "message-createChat": creates a chat
    "message-delChat": deletes the group
    "message-getChat": gets information about the chat
    "message-send": sends a text to the chat
    "message-checkGroup": check if the groups exists

steps to use as client:
connect to websocket
    send "message-getChat"
        send JSON {"username":username, "password":password,"cId":chat id}
        if valid information
            recieve JSON chat dicionary
        else
            recieve "failed"
            

    send "message-createChat"
        send JSON {"username":username, "password":password,"displayName":chat name,"owner",owner username}
        if valid login 
            recieve "success"
        else
            recieve "failed"
    
    send "message-delChat"
        send JSON {"username":username, "password":password,"displayName":chat name,"owner",owner username}
        if valid login 
            recieve "success"
        else
            recieve "failed"

    send "message-send"
        send JSON {"cId": chat id, "username":username, "password":password, "text": the message to send}
        if all info valid:
            recieve "success"
        if not owner
            recieve "Not the owner"
        else 
            recieve "Failed!"
        

    send "message-addUser"
        send JSON {"cId": chat id, "newUser": the username of the new user, "username": owners username,"password":owners password}
        if info valid
            recieve "success"
        else
            recieve "failed"

    send "message-checkGroup"
        send cId
        if group exists
            recieve "success"
        else
            recieve "failed"
